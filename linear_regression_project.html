def matxMultiply(A, B):
    if len(A[0]) != len(B):
        raise ValueError
    res = []
    for i in range(len(A)):
        row = []
        for k in range(len(B[0])):
            tmp = 0
            for j in range(len(A[0])):
                tmp += A[i][j] * B[j][k]
            row.append(tmp)
        res.append(row)
    return res

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    def isZero(x, epsilon):
        if -epsilon <= x <= epsilon:
            return True
        return False
    # step1 A, b
    if len(A) != len(b):
        return None
    # step2 
    augMax = augmentMatrix(A, b)
    # step3 
    # rows cloumns
    rows = len(A)
    columns = len(A[0])
    # 
    for j in range(columns):
        mx = 0
        mxi = -1
        i = j
        while i < rows:
            if abs(A[i][j]) > mx:
                mx = abs(A[i][j])
                mxi = i
            i += 1
        # 
        if isZero(mx, epsilon):
            return None
        # 
        swapRows(augMax, j, mxi)
        # 
        if not isZero(augMax[j][j], epsilon):
            scale = 1.0 / augMax[j][j]
            scaleRow(augMax, j, scale)
        for i in range(rows):
            if i == j:
                continue
            scale1 = augMax[i][j] * (-1)
            addScaledRow(augMax, i, j, scale1)

    res = []
    for i in range(rows):
        tmp = [augMax[i][columns]]
        res.append(tmp)
    matxRound(res, decPts)
    return res

def calculateMSE2D(X,Y,m,b):
    n = len(X)
    sum = 0
    for i in range(n):
        sum += (Y[i] - m * X[i] - b) ** 2
    sum /= n
    return sum
